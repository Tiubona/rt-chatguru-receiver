      <label>Token (x-rt-admin-token)</label>

      <input id="token" placeholder="Cole o token aqui" />

        <button class="btn" id="btn">Entrar</button>

        <small id="msg"></small>

  </div>


  <script>

    const $ = (id) => document.getElementById(id);

    const token = localStorage.getItem("rt_admin_token");

    if (token) $("token").value = token;


    $("btn").onclick = () => {

      const t = $("token").value.trim();

      if (!t) { $("msg").textContent = "Cole o token."; return; }

      localStorage.setItem("rt_admin_token", t);

      window.location.href = "/admin";

    };

  </script>
